# Команды для обновления Aero Lunch Bot на сервере
# Выполните эти команды по порядку после подключения к серверу

# 1. Перейти в директорию проекта
cd /root/food-telegram-app

# 2. Получить последние изменения из Git
git pull origin main

# 3. Обновить зависимости фронтенда
npm install
npm run build

# 4. Обновить бэкенд
cd server
npm install
npx prisma generate
npx prisma db push

# 5. Перезапустить сервисы
pm2 restart frontend
pm2 restart backend

# 6. Проверить статус
pm2 status

# 7. Проверить логи (если нужно)
pm2 logs frontend --lines 20
pm2 logs backend --lines 20

echo "✅ Обновление завершено! Теперь в боте доступен личный кабинет с профилем, бонусами и историей заказов."
